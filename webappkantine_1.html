<script>

    // Madplan med billed URLs (kan ændres til rigtige billeder)

    const meals = [

        { day: 'Mandag', name: 'Butter Chicken med blomkål & kikærter', nutrition: 'Protein: 25g, Kalorier: 450', allergy: 'Indeholder nødder', img: 'https://i.imgur.com/EqjYNgG.jpg' },

        { day: 'Tirsdag', name: 'Vegetarisk lasagne', nutrition: 'Protein: 15g, Kalorier: 400', allergy: 'Gluten, mælk', img: 'https://i.imgur.com/CnpJywV.jpg' },

        { day: 'Onsdag', name: 'Grøntsagswraps', nutrition: 'Protein: 12g, Kalorier: 350', allergy: 'Ingen', img: 'https://i.imgur.com/tIYVHZr.jpg' },

        { day: 'Torsdag', name: 'Laks med quinoa og grøntsager', nutrition: 'Protein: 30g, Kalorier: 500', allergy: 'Fisk', img: 'https://i.imgur.com/Aa7oFVb.jpg' },

        { day: 'Fredag', name: 'Pizza med frisk tomat og basilikum', nutrition: 'Protein: 20g, Kalorier: 600', allergy: 'Gluten, mælk', img: 'https://i.imgur.com/AqsIlNj.jpg' }

    ];

    // Drikkevarer med billeder (kan erstattes af egne billeder)

    const drinks = [

        { name: 'Danskvand', price: 8, img: 'https://i.imgur.com/vcgV47R.png' },

        { name: 'Mineralvand', price: 8, img: 'https://i.imgur.com/yTchUp9.png' },

        { name: 'Æblejuice', price: 10, img: 'https://i.imgur.com/5Qwyv4X.png' },

        { name: 'Iste', price: 9, img: 'https://i.imgur.com/Rlf8oQb.png' }

    ];

    const planDiv = document.getElementById('plan');

    const freeDrinksDiv = document.getElementById('freeDrinks');

    let orders = JSON.parse(localStorage.getItem('orders')) || {};

    let freeDrinksOrders = JSON.parse(localStorage.getItem('freeDrinksOrders')) || {};

    let feedbacks = JSON.parse(localStorage.getItem('feedbacks')) || {};

    let userName = localStorage.getItem('userName') || '';

    // Navne-input øverst

    const nameLabel = document.createElement('label');

    nameLabel.htmlFor = 'userNameInput';

    nameLabel.textContent = 'Navn på bestilling:';

    nameLabel.style.marginTop = '20px';

    planDiv.parentNode.insertBefore(nameLabel, planDiv);

    const nameInput = document.createElement('input');

    nameInput.type = 'text';

    nameInput.id = 'userNameInput';

    nameInput.placeholder = 'Indtast dit navn her';

    nameInput.value = userName;

    nameInput.oninput = () => {

        userName = nameInput.value.trim();

        localStorage.setItem('userName', userName);

    };

    planDiv.parentNode.insertBefore(nameInput, planDiv);

    // Opret inputfelter til portionsnavne og drikkevarer med billeder

    function createPortionInputs(day) {

        const container = document.createElement('div');

        container.className = 'portion-section';

        const savedPortions = orders[day]?.portions || [];

        savedPortions.forEach((portion, idx) => {

            const nameLabel = document.createElement('label');

            nameLabel.textContent = `Navn på portion ${idx + 1}:`;

            const nameInput = document.createElement('input');

            nameInput.type = 'text';

            nameInput.value = portion.name || '';

            nameInput.className = 'name-input';

            // Drikkevarevælger med billeder

            const drinkLabel = document.createElement('label');

            drinkLabel.textContent = `Vælg drikkevare til portion ${idx + 1}:`;

            drinkLabel.className = 'drink-label';

            const drinkSelect = document.createElement('select');

            drinkSelect.style.paddingLeft = '10px';

            const emptyOption = document.createElement('option');

            emptyOption.value = '';

            emptyOption.textContent = 'Ingen drikkevare';

            drinkSelect.appendChild(emptyOption);

            drinks.forEach(drink => {

                const opt = document.createElement('option');

                opt.value = drink.name;

                opt.textContent = drink.name;

                if (portion.drink === drink.name) opt.selected = true;

                drinkSelect.appendChild(opt);

            });

            // Opdater data ved ændringer

            nameInput.oninput = () => {

                orders[day].portions[idx].name = nameInput.value;

                localStorage.setItem('orders', JSON.stringify(orders));

            };

            drinkSelect.onchange = () => {

                orders[day].portions[idx].drink = drinkSelect.value;

                localStorage.setItem('orders', JSON.stringify(orders));

            };

            container.appendChild(nameLabel);

            container.appendChild(nameInput);

            container.appendChild(drinkLabel);

            container.appendChild(drinkSelect);

        });

        return container;

    }

    meals.forEach(meal => {

        const dayDiv = document.createElement('div');

        dayDiv.className = 'day';

        const mealTitle = document.createElement('div');

        mealTitle.className = 'meal-title';

        // Billede ved siden af rettens navn

        const mealImg = document.createElement('img');

        mealImg.src = meal.img;

        mealImg.alt = meal.name;

        mealImg.loading = 'lazy';

        const mealText = document.createElement('span');

        mealText.textContent = `${meal.day}: ${meal.name}`;

        mealTitle.appendChild(mealImg);

        mealTitle.appendChild(mealText);

        dayDiv.appendChild(mealTitle);

        const nutritionDiv = document.createElement('div');

        nutritionDiv.className = 'info';

        nutritionDiv.innerHTML = `<strong>Næringsindhold:</strong> ${meal.nutrition}`;

        dayDiv.appendChild(nutritionDiv);

        const allergyDiv = document.createElement('div');

        allergyDiv.className = 'info';

        allergyDiv.innerHTML = `<strong>Allergi info:</strong> ${meal.allergy}`;

        dayDiv.appendChild(allergyDiv);

        // Antalsvalg af portioner

        const qtyLabel = document.createElement('label');

        qtyLabel.htmlFor = 'qty-' + meal.day;

        qtyLabel.textContent = 'Antal portioner (0-5):';

        const qtyInput = document.createElement('input');

        qtyInput.type = 'number';

        qtyInput.min = 0;

        qtyInput.max = 5;

        qtyInput.id = 'qty-' + meal.day;

        qtyInput.value = orders[meal.day]?.portions?.length || 0;

        // Dynamisk container til portionsnavne og drikkevarer

        let portionsContainer = createPortionInputs(meal.day);

        qtyInput.onchange = () => {

            let val = parseInt(qtyInput.value);

            if (isNaN(val) || val < 0) val = 0;

            else if (val > 5) val = 5;

            qtyInput.value = val;

            if (!orders[meal.day]) orders[meal.day] = { portions: [] };

            const currentLen = orders[meal.day].portions.length;

            if (val > currentLen) {

                for (let i = currentLen; i < val; i++) {

                    orders[meal.day].portions.push({ name: '', drink: '' });

                }

            } else if (val < currentLen) {

                orders[meal.day].portions = orders[meal.day].portions.slice(0, val);

            }

            localStorage.setItem('orders', JSON.stringify(orders));

            dayDiv.removeChild(portionsContainer);

            portionsContainer = createPortionInputs(meal.day);

            dayDiv.appendChild(portionsContainer);

        };

        const orderButton = document.createElement('button');

        orderButton.textContent = 'Gem bestilling';

        orderButton.onclick = () => {

            if (userName === '') {

                alert('Indtast venligst navn på bestillingen.');

                return;

            }

            const val = parseInt(qtyInput.value);

            if (isNaN(val) || val < 0) {

                alert('Indtast venligst et gyldigt antal portioner.');

                return;

            }

            if (val === 0) {

                delete orders[meal.day];

            } else {

                if (!orders[meal.day]) orders[meal.day] = { portions: [] };

                if (orders[meal.day].portions.length !== val) {

                    orders[meal.day].portions.length = val;

                }

            }

            localStorage.setItem('orders', JSON.stringify(orders));

            orderStatus.textContent = `Bestilling gemt for ${meal.day}`;

        };

        const orderStatus = document.createElement('div');

        orderStatus.className = 'status';

        // Feedback-sektion

        const feedbackLabel = document.createElement('label');

        feedbackLabel.htmlFor = 'feedback-' + meal.day;

        feedbackLabel.textContent = 'Feedback på ' + meal.day + ':';

        const feedbackTextarea = document.createElement('textarea');

        feedbackTextarea.id = 'feedback-' + meal.day;

        feedbackTextarea.rows = 3;

        feedbackTextarea.value = feedbacks[meal.day] || '';

        const feedbackButton = document.createElement('button');

        feedbackButton.textContent = 'Send Feedback';

        feedbackButton.onclick = () => {

            const text = feedbackTextarea.value.trim();

            if (text === '') {

                alert('Skriv venligst feedback for ' + meal.day);

                return;

            }

            feedbacks[meal.day] = text;

            localStorage.setItem('feedbacks', JSON.stringify(feedbacks));

            feedbackStatus.textContent = 'Feedback sendt for ' + meal.day;

        };

        const feedbackStatus = document.createElement('div');

        feedbackStatus.className = 'status';

        dayDiv.appendChild(qtyLabel);

        dayDiv.appendChild(qtyInput);

        dayDiv.appendChild(portionsContainer);

        dayDiv.appendChild(orderButton);

        dayDiv.appendChild(orderStatus);

        dayDiv.appendChild(feedbackLabel);

        dayDiv.appendChild(feedbackTextarea);

        dayDiv.appendChild(feedbackButton);

        dayDiv.appendChild(feedbackStatus);

        planDiv.appendChild(dayDiv);

    });

    // Sektion for drikkevarer til fri bestilling, inkl. billeder

    freeDrinksDiv.innerHTML = '<h3>Bestil drikkevarer separat</h3>';

    drinks.forEach(drink => {

        const drinkDiv = document.createElement('div');

        drinkDiv.style.marginBottom = '20px';

        const drinkLabel = document.createElement('label');

        drinkLabel.className = 'drink-label';

        const drinkImg = document.createElement('img');

        drinkImg.src = drink.img;

        drinkImg.alt = drink.name;

        drinkImg.loading = 'lazy';

        drinkLabel.appendChild(drinkImg);

        drinkLabel.appendChild(document.createTextNode(`${drink.name} - Pris: ${drink.price} kr`));

        const drinkQtyInput = document.createElement('input');

        drinkQtyInput.type = 'number';

        drinkQtyInput.min = 0;

        drinkQtyInput.max = 20;

        drinkQtyInput.id = 'freeDrinkQty-' + drink.name;

        drinkQtyInput.value = freeDrinksOrders[drink.name]?.qty || 0;

        const drinkOrderButton = document.createElement('button');

        drinkOrderButton.textContent = 'Gem drikkevare-bestilling';

        drinkOrderButton.onclick = () => {

            if (userName === '') {

                alert('Indtast venligst navn på bestillingen.');

                return;

            }

            const qty = parseInt(drinkQtyInput.value);

            if (isNaN(qty) || qty < 0) {

                alert('Indtast venligst et gyldigt antal.');

                return;

            }

            if (qty === 0) {

                delete freeDrinksOrders[drink.name];

            } else {

                freeDrinksOrders[drink.name] = { qty: qty };

            }

            localStorage.setItem('freeDrinksOrders', JSON.stringify(freeDrinksOrders));

            freeDrinkStatus.textContent = `Bestilling gemt for ${drink.name}`;

        };

        const freeDrinkStatus = document.createElement('div');

        freeDrinkStatus.className = 'status';

        drinkDiv.appendChild(drinkLabel);

        drinkDiv.appendChild(drinkQtyInput);

        drinkDiv.appendChild(drinkOrderButton);

        drinkDiv.appendChild(freeDrinkStatus);

        freeDrinksDiv.appendChild(drinkDiv);

    });
